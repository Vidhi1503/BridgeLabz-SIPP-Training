import org.junit.jupiter.api.Test;
import static org.junit.jupiter.api.Assertions.*;

class PasswordValidator {
    static boolean isValid(String pwd) {
        return pwd.length() >= 8 && pwd.matches(".*[A-Z].*") && pwd.matches(".*\\d.*");
    }
}

class PasswordValidatorTest {
    @Test
    void testValidPassword() {
        assertTrue(PasswordValidator.isValid("Strong1Pass"), "Valid password should return true");
    }

    @Test
    void testInvalidPassword() {
        assertFalse(PasswordValidator.isValid("weak"), "Invalid password should return false");
    }
}
