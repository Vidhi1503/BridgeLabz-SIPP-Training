import org.junit.jupiter.api.Test;
import static org.junit.jupiter.api.Assertions.*;

class TemperatureConverter {
    static double celsiusToFahrenheit(double c) { return c * 9/5 + 32; }
    static double fahrenheitToCelsius(double f) { return (f - 32) * 5/9; }
}

class TemperatureConverterTest {
    @Test
    void testCtoF() {
        assertEquals(212, TemperatureConverter.celsiusToFahrenheit(100), 0.01,
                     "Celsius to Fahrenheit conversion failed");
    }

    @Test
    void testFtoC() {
        assertEquals(0, TemperatureConverter.fahrenheitToCelsius(32), 0.01,
                     "Fahrenheit to Celsius conversion failed");
    }
}
