import org.junit.jupiter.api.Test;
import static org.junit.jupiter.api.Assertions.*;

class BankAccount {
    private double balance = 0;

    void deposit(double amount) { balance += amount; }

    void withdraw(double amount) {
        if (amount > balance) throw new IllegalArgumentException("Insufficient funds");
        balance -= amount;
    }

    double getBalance() { return balance; }
}

class BankAccountTest {
    @Test
    void testDeposit() {
        BankAccount acc = new BankAccount();
        acc.deposit(100);
        assertEquals(100, acc.getBalance(), "Balance after deposit should be correct");
    }

    @Test
    void testWithdraw() {
        BankAccount acc = new BankAccount();
        acc.deposit(100);
        acc.withdraw(40);
        assertEquals(60, acc.getBalance(), "Balance after withdrawal should be correct");
    }

    @Test
    void testWithdrawInsufficient() {
        BankAccount acc = new BankAccount();
        assertThrows(IllegalArgumentException.class,
            () -> acc.withdraw(50),
            "Should throw for insufficient funds");
    }
}
